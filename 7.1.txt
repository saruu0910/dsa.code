def binary_search(arr, key):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == key:
            return f"Element {key} found at index {mid}"
        elif arr[mid] < key:
            low = mid + 1
        else:
            high = mid - 1
    return "Element not found"

nums = [10, 20, 30, 40, 50]
print(binary_search(nums, 40))
