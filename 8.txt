class Queue:
    def __init__(self):
        self.q = []

    def enqueue(self, item):
        self.q.append(item)

    def dequeue(self):
        return self.q.pop(0)

Q = Queue()
Q.enqueue(10)
Q.enqueue(20)
Q.enqueue(30)
print("Queue:", Q.q)
Q.dequeue()
print("After dequeue:", Q.q)
